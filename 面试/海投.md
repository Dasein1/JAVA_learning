class Solution{
    public int n;
    public String s;
    public String getResult(n,s){
        int cur_num=0;
        int start_index=0;
        int end_index=0;
        int state=0; //0 英文  1数字
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)>='0' && s.charAt(i)<=9){
                    if(state==0){
                        state=1;
                        cur_num++;
                        if(cur_num==this.n){
                            start_index=i;
                        }
                    }
                }
            else{
               if(state==1){
                    state=0;
                    if(cur_num==this.n){
                        end_index=i;
                        break;
                    }
               }
            }
        }
        return s.substring(start_index,end_index);
    }
}
